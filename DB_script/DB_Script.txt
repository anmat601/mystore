 ** ITEM_DATA TABLE
 
1. TABLE CREATION
	
CREATE TABLE ITEM_DATA(
	ITEM_NAME VARCHAR(15) NOT NULL,
	ITEM_PRICE INT NOT NULL,
	ITEM_MEASUREMENT VARCHAR(15) NOT NULL,
	PRIMARY KEY(ITEM_NAME));

**STOCK_DETAILS TABLE
 
1.TABLE CREATION
	
CREATE TABLE STOCK_DETAILS(
	ITEM_NAME VARCHAR(15) NOT NULL,
	AVAILABLE_STOCK INT ,
	MIN_EXPECTED_STOCK INT,
	SUPPLIER VARCHAR(50),
	FOREIGN KEY(ITEM_NAME) REFERENCES ITEM_DATA(ITEM_NAME) );
